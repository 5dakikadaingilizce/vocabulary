/* ── GÜNCELLENMİŞ SÖZLÜK & DERS NOTU FONKSİYONU ── */ function renderNotesTab(tab) { const body = document.getElementById('notes-body'); // 1. KONTROL: Eğer gelen veri bir kelime (vocabulary) ise tabloyu bas if (tab.type === 'vocabulary') { body.innerHTML = `

${tab.word.toUpperCase()}

Form Kelime Türkçesi
V1${tab.v1}${tab.tr_v1 || ''}
V2${tab.v2}${tab.tr_v2 || ''}
V3${tab.v3}${tab.tr_v3 || ''}
ING${tab.ing}${tab.tr_ing || ''}

İsim (Noun): ${tab.noun}

Sıfat (Adj): ${tab.adj}

Zarf (Adv): ${tab.adv}

`; } // 2. KONTROL: Eğer standart ders notu ise eski yapıyı çalıştır else { const imgHtml = tab.image ? `

🔍 Tam boy görmek için resme tıkla

` : ''; body.innerHTML = imgHtml + (tab.items||[]).map(item => `
${item.tag||''}
${item.text||''}
`).join(''); } }
